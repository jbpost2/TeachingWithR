  • Often want to execute statements conditionally to create a variable

  then execute code
} 

#if then else
if (condition) {
  execute this code  
} else {
  execute this code
}
#Or more if statements
if (condition) {
  execute this code  
} else if (condition2) {
  execute this code
} else if (condition3) {
  execute this code
} else {
  #if no conditions met
  execute this code
}
